import xml.etree.ElementTree as ET

def split_xml(input_file, items_per_file=1000):
    """–†–∞–∑–±–∏–≤–∞–µ—Ç XML —Ñ–∞–π–ª –Ω–∞ —á–∞—Å—Ç–∏."""

    # –ù–∞—Ö–æ–¥–∏–º namespace
    namespace = None

        namespace = root.tag[1:root.tag.index('}')]
    
    # –ù–∞—Ö–æ–¥–∏–º –∫–∞—Ç–∞–ª–æ–≥ –∏ —Ç–æ–≤–∞—Ä—ã
    catalog = root.find(f'.//{{{namespace}}}–ö–∞—Ç–∞–ª–æ–≥') if namespace else root.find('.//–ö–∞—Ç–∞–ª–æ–≥')
    if catalog is None:
        print("–ö–∞—Ç–∞–ª–æ–≥ –Ω–µ –Ω–∞–π–¥–µ–Ω!")
        return
    
    products = catalog.findall(f'.//{{{namespace}}}–¢–æ–≤–∞—Ä') if namespace else catalog.findall('.//–¢–æ–≤–∞—Ä')
    print(f"–ù–∞–π–¥–µ–Ω–æ —Ç–æ–≤–∞—Ä–æ–≤: {len(products)}")
    
    # –†–∞–∑–±–∏–≤–∞–µ–º –Ω–∞ —á–∞—Å—Ç–∏
    for i in range(0, len(products), items_per_file):

        output_file = f"{input_file}.part{part_num}.xml"
        
        # –°–æ–∑–¥–∞–µ–º –Ω–æ–≤—ã–π –∫–æ—Ä–Ω–µ–≤–æ–π —ç–ª–µ–º–µ–Ω—Ç

        # –ö–æ–ø–∏—Ä—É–µ–º –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ç–æ—Ä (–µ—Å–ª–∏ –µ—Å—Ç—å)

        if classifier is not None:
            new_root.append(classifier)
        
        # –°–æ–∑–¥–∞–µ–º –Ω–æ–≤—ã–π –∫–∞—Ç–∞–ª–æ–≥
        new_catalog = ET.SubElement(new_root, catalog.tag, catalog.attrib)
        
        # –ö–æ–ø–∏—Ä—É–µ–º –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—é –æ –∫–∞—Ç–∞–ª–æ–≥–µ

                new_products = ET.SubElement(new_catalog, child.tag)
                # –î–æ–±–∞–≤–ª—è–µ–º —Ç–æ–≤–∞—Ä—ã –¥–ª—è —ç—Ç–æ–π —á–∞—Å—Ç–∏
                for product in products[i:i+items_per_file]:
                    new_products.append(product)
            else:
                new_catalog.append(child)
        
        # –°–æ—Ö—Ä–∞–Ω—è–µ–º
        new_tree = ET.ElementTree(new_root)
        ET.indent(new_tree, space="  ")
        new_tree.write(output_file, encoding='utf-8', xml_declaration=True)
        print(f"–°–æ–∑–¥–∞–Ω —Ñ–∞–π–ª: {output_file} ({len(products[i:i+items_per_file])} —Ç–æ–≤–∞—Ä–æ–≤)")

    split_xml(input_file, items_per_file)
